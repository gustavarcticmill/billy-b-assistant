def _float_env(key: str, default: str) -> float:
    value = os.getenv(key)
    if value is None:
        return float(default)
    try:
        return float(value)
    except (TypeError, ValueError):
        print(
            f"Warning: Invalid float for {key}={value!r}, falling back to {default}",
            flush=True,
        )
        return float(default)
